# Trying Spark Standalone (i. e. without Kubernetes) locally
## Install Spark locally
```sh
brew install apache-spark

which pyspark
realpath $(which pyspark)
dirname $(realpath $(which pyspark))

# Get an overview over the Spark dir
export SPARK_HOME=$(dirname $(realpath $(which pyspark)))
ls -lah $SPARK_HOME

# For a Python Spark Shell
pyspark

# For a Python Jupyter Notebook with Spark Support
export PYSPARK_DRIVER_PYTHON=jupyter
export PYSPARK_DRIVER_PYTHON_OPTS='notebook'
pyspark
```


## Run PySpark from Notebook
```py
import pyspark

# pyspark will launch the notebook process with the env var PYTHONPATH incl. `$SPARK_HOME/python`
print(pyspark.__file__)
```
